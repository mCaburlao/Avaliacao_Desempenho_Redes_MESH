# 5 STAs — UDP echo + 1 TCP file transfer
# wiredSTA IPs: 10.4.128.1-5  |  Backhaul: 10.1.1.1
# STAs escalonadas 5s entre si | Interval=0.5s | 100 pacotes = 50s/STA

# --- Probes: forcam route discovery do AODV (excluidos da analise) ---
defaults StartTime=240s StopTime=269s
probe1-cl    10.4.128.1    echo_client
probe1-srv   10.1.1.1    echo_server

probe2-cl    10.4.128.2    echo_client
probe2-srv   10.1.1.1    echo_server

probe3-cl    10.4.128.3    echo_client
probe3-srv   10.1.1.1    echo_server

probe4-cl    10.4.128.4    echo_client
probe4-srv   10.1.1.1    echo_server

probe5-cl    10.4.128.5    echo_client
probe5-srv   10.1.1.1    echo_server

# Conexoes probe
connect probe1-cl    probe1-srv
connect probe2-cl    probe2-srv
connect probe3-cl    probe3-srv
connect probe4-cl    probe4-srv
connect probe5-cl    probe5-srv

# --- Fluxos de medicao (t >= warmup_s) ---
# Interval=0.5s e MaxPackets=100 definidos inline no echo_client
# (UdpEchoServer nao tem esses atributos — nao usar defaults)
defaults StopTime=
defaults StartTime=270s
echo1-cl    10.4.128.1    echo_client    Interval=0.5s MaxPackets=100
echo1-srv   10.1.1.1    echo_server

defaults StartTime=275s
echo2-cl    10.4.128.2    echo_client    Interval=0.5s MaxPackets=100
echo2-srv   10.1.1.1    echo_server

defaults StartTime=280s
echo3-cl    10.4.128.3    echo_client    Interval=0.5s MaxPackets=100
echo3-srv   10.1.1.1    echo_server

defaults StartTime=285s
echo4-cl    10.4.128.4    echo_client    Interval=0.5s MaxPackets=100
echo4-srv   10.1.1.1    echo_server

defaults StartTime=290s
echo5-cl    10.4.128.5    echo_client    Interval=0.5s MaxPackets=100
echo5-srv   10.1.1.1    echo_server

# TCP file transfer: backhaul -> STA mais distante (max hops)
defaults StartTime=295s
file5-srv   10.1.1.1    file_server
file5-cl    10.4.128.5    client

# Conexoes medicao
connect echo1-cl    echo1-srv
connect echo2-cl    echo2-srv
connect echo3-cl    echo3-srv
connect echo4-cl    echo4-srv
connect echo5-cl    echo5-srv
connect file5-srv   file5-cl
